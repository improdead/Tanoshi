# Google Colab Setup for AI Manga Analysis Backend (MagiV2 + XTTS‑v2)

## AUTOMATIC DATA FLOW OVERVIEW

**How manga gets from iOS app to Colab automatically:**

1. **PDF Processing (iOS)**: When user opens a manga chapter, the app uses PDFKit to extract each page as a UIImage
2. **Image Conversion (iOS)**: Each UIImage is converted to PNG data, then base64-encoded for HTTP transmission
3. **HTTP Request (iOS → Colab)**: App sends POST request to `/analyze` with array of base64 images
4. **Image Decoding (Colab)**: Server decodes base64 back to PIL Images, converts to numpy arrays
5. **AI Processing (Colab)**: MagiV2 processes all pages with character bank for OCR + character detection
6. **JSON Response (Colab → iOS)**: Results sent back as structured JSON with dialogue and character data
7. **Local Caching (iOS)**: App stores results locally for offline access and immediate loading

**Example Data Structure Sent to Colab:**
```json
{
  "pages": [
    "iVBORw0KGgoAAAANSUhEUgAA...", // base64 page 1
    "iVBORw0KGgoAAAANSUhEUgAA...", // base64 page 2
    // ... more pages
  ],
  "characterBank": {
    "images": ["iVBORw0KGgoAAAANSUhEUgAA..."], // base64 character refs
    "names": ["Luffy", "Zoro", "Sanji"]
  }
}
```

```python
# Cell 5: Utility Functions
def read_image_from_base64(base64_string):
    """Convert base64 string to PIL Image"""
    try:
        image_data = base64.b64decode(base64_string)
        image = Image.open(BytesIO(image_data)).convert("L").convert("RGB")
        return np.array(image)
    except Exception as e:
        print(f"Error reading image from base64: {e}")
        return None

def image_to_base64(image_array):
    """Convert numpy array to base64 string"""
    try:
        image = Image.fromarray(image_array.astype(np.uint8))
        buffer = BytesIO()
        image.save(buffer, format='PNG')
        return base64.b64encode(buffer.getvalue()).decode()
    except Exception as e:
        print(f"Error converting image to base64: {e}")
        return None

def create_job_id():
    """Generate unique job ID"""
    return str(uuid.uuid4())

def convert_bbox(bbox_coords):
    """Convert MagiV2 bbox format to iOS expected format"""
    if not bbox_coords or len(bbox_coords) < 4:
        return {"x": 0, "y": 0, "width": 0, "height": 0}
    
    return {
        "x": float(bbox_coords[0]),
        "y": float(bbox_coords[1]), 
        "width": float(bbox_coords[2] - bbox_coords[0]),
        "height": float(bbox_coords[3] - bbox_coords[1])
    }

class AnalysisJob:
    def __init__(self, job_id: str):
        self.job_id = job_id
        self.status = "pending"  # pending, processing, completed, failed
        self.progress = 0.0
        self.result = None
        self.error = None
        self.created_at = time.time()
        
    def to_dict(self):
        return {
            "job_id": self.job_id,
            "status": self.status,
            "progress": self.progress,
            "result": self.result,
            "error": self.error,
            "created_at": self.created_at
        }

print("Utility functions defined!")
```

```python
# Cell 6: MagiV2 Analysis Functions
def process_manga_analysis(job_id: str, chapter_pages: List[np.ndarray], character_bank: Dict[str, Any]):
    """Process manga analysis using MagiV2 model with proper character bank integration"""
    job = analysis_jobs[job_id]
    
    try:
        job.status = "processing"
        job.progress = 0.1
        
        print(f"Starting analysis for job {job_id} with {len(chapter_pages)} pages")
        
        # Prepare character bank for MagiV2
        character_images = []
        character_names = character_bank.get("names", [])
        
        for img_b64 in character_bank.get("images", []):
            img = read_image_from_base64(img_b64)
            if img is not None:
                character_images.append(img)
        
        character_bank_processed = {
            "images": character_images,
            "names": character_names
        }
        
        job.progress = 0.2
        
        # Run MagiV2 analysis with character bank
        with torch.no_grad():
            per_page_results = magi_model.do_chapter_wide_prediction(
                chapter_pages, 
                character_bank_processed, 
                use_tqdm=True, 
                do_ocr=True
            )
        
        job.progress = 0.8
        
        # Process results exactly like the original MagiV2 example
        transcript = []
        pages_analysis = []
        
        for i, (image, page_result) in enumerate(zip(chapter_pages, per_page_results)):
            # Create speaker name mapping
            speaker_name = {
                text_idx: page_result["character_names"][char_idx] 
                for text_idx, char_idx in page_result["text_character_associations"]
            }
            
            # Process text regions
            text_regions = []
            for j, text in enumerate(page_result["ocr"]):
                if page_result["is_essential_text"][j]:
                    # Get bounding box if available
                    bbox = {}
                    if "text_bboxes" in page_result and j < len(page_result["text_bboxes"]):
                        bbox = convert_bbox(page_result["text_bboxes"][j])
                    
                    text_regions.append({
                        "id": j,
                        "text": text,
                        "boundingBox": bbox,
                        "confidence": 1.0,
                        "isEssential": True
                    })
                    
                    # Add to transcript with proper speaker
                    speaker = speaker_name.get(j, "unknown")
                    transcript.append({
                        "pageIndex": i,
                        "textId": j,
                        "speaker": speaker,
                        "text": text,
                        "timestamp": None
                    })
            
            # Process character detections
            character_detections = []
            for char_idx, char_name in enumerate(page_result["character_names"]):
                bbox = {}
                if "character_bboxes" in page_result and char_idx < len(page_result["character_bboxes"]):
                    bbox = convert_bbox(page_result["character_bboxes"][char_idx])
                
                character_detections.append({
                    "id": char_idx,
                    "name": char_name,
                    "boundingBox": bbox,
                    "confidence": 1.0
                })
            
            pages_analysis.append({
                "pageIndex": i,
                "textRegions": text_regions,
                "characterDetections": character_detections,
                "textCharacterAssociations": page_result["text_character_associations"]
            })
            
            # Generate visualization for debugging
            magi_model.visualise_single_image_prediction(
                image, page_result, f"/tmp/analysis_{job_id}_page_{i}.png"
            )
        
        # Create final result matching iOS expected format
        result = {
            "pages": pages_analysis,
            "transcript": transcript,
            "analysisDate": time.time(),
            "version": "1.0"
        }
        
        job.result = result
        job.status = "completed"
        job.progress = 1.0
        
        print(f"Analysis completed for job {job_id}")
        
    except Exception as e:
        job.status = "failed"
        job.error = str(e)
        print(f"Analysis failed for job {job_id}: {e}")
        import traceback
        traceback.print_exc()

print("MagiV2 analysis functions defined!")
```

```python
# Cell 7: Coqui XTTS-v2 Audio Generation with Emotional Voice Variation
def generate_audio_for_transcript(job_id: str, transcript: List[Dict], voice_settings: Dict[str, Any]):
    """Generate audio for transcript using XTTS-v2 with voice cloning and emotional variation"""
    job = audio_jobs[job_id]
    
    try:
        job.status = "processing"
        job.progress = 0.1
        
        print(f"Starting audio generation for job {job_id} with {len(transcript)} dialogue lines")
        
        # Get TTS model
        tts_model = tts_models.get('xtts_v2')
        if not tts_model:
            raise Exception("XTTS-v2 model not loaded")
        
        # Voice settings
        language = voice_settings.get("language", "en")
        character_voice_files = voice_settings.get("characterVoiceFiles", {})  # character_name: base64_audio_file
        default_voice_file = voice_settings.get("defaultVoiceFile", None)  # base64 audio file
        enable_emotional_variation = voice_settings.get("enableEmotionalVariation", False)
        
        audio_segments = []
        
        for i, dialogue in enumerate(transcript):
            try:
                speaker = dialogue["speaker"]
                text = dialogue["text"]
                emotion = dialogue.get("emotion", "neutral")
                
                # Skip empty text
                if not text.strip():
                    continue
                
                # Determine voice file to use for cloning
                voice_file_b64 = character_voice_files.get(speaker, default_voice_file)
                
                # Generate audio with emotional variation
                audio_path = f"/tmp/audio_{job_id}_{i}.wav"
                
                # Apply emotional modifications to text if enabled
                if enable_emotional_variation:
                    text = apply_emotional_text_modification(text, emotion)
                
                if voice_file_b64:
                    # Use voice cloning with provided speaker sample
                    speaker_wav_path = f"/tmp/speaker_{job_id}_{i}.wav"
                    
                    # Decode base64 voice file
                    voice_data = base64.b64decode(voice_file_b64)
                    with open(speaker_wav_path, "wb") as f:
                        f.write(voice_data)
                    
                    # Generate speech with voice cloning and emotional parameters
                    generation_kwargs = {
                        "text": text,
                        "file_path": audio_path,
                        "speaker_wav": speaker_wav_path,
                        "language": language
                    }
                    
                    # Add emotional parameters if supported
                    if enable_emotional_variation:
                        generation_kwargs.update(get_emotional_parameters(emotion))
                    
                    tts_model.tts_to_file(**generation_kwargs)
                    
                    # Clean up speaker file
                    os.remove(speaker_wav_path)
                    
                else:
                    # Use default built-in speaker if available
                    available_speakers = tts_model.speakers if hasattr(tts_model, 'speakers') else []
                    generation_kwargs = {
                        "text": text,
                        "file_path": audio_path,
                        "language": language
                    }
                    
                    if available_speakers:
                        generation_kwargs["speaker"] = available_speakers[0]
                    
                    # Add emotional parameters if supported
                    if enable_emotional_variation:
                        generation_kwargs.update(get_emotional_parameters(emotion))
                    
                    tts_model.tts_to_file(**generation_kwargs)
                
                # Post-process audio for emotional effect if needed
                if enable_emotional_variation and emotion != "neutral":
                    audio_path = apply_audio_emotional_effects(audio_path, emotion, job_id, i)
                
                # Read audio file and convert to base64
                with open(audio_path, "rb") as f:
                    audio_data = base64.b64encode(f.read()).decode()
                
                # Calculate duration using librosa
                import librosa
                y, sr = librosa.load(audio_path)
                duration = librosa.get_duration(y=y, sr=sr)
                
                audio_segments.append({
                    "dialogueId": f"{dialogue['pageIndex']}_{dialogue['textId']}",
                    "audioData": audio_data,
                    "duration": duration,
                    "speaker": speaker,
                    "text": dialogue["text"],  # Original text
                    "emotion": emotion
                })
                
                # Clean up temp file
                os.remove(audio_path)
                
                job.progress = 0.1 + (0.8 * (i + 1) / len(transcript))
                
            except Exception as e:
                print(f"Failed to generate audio for dialogue {i}: {e}")
                continue
        
        job.result = {"audioSegments": audio_segments}
        job.status = "completed"
        job.progress = 1.0
        
        print(f"Audio generation completed for job {job_id}")
        
    except Exception as e:
        job.status = "failed"
        job.error = str(e)
        print(f"Audio generation failed for job {job_id}: {e}")

def apply_emotional_text_modification(text: str, emotion: str) -> str:
    """Modify text based on emotion for better TTS output"""
    if emotion == "shouting":
        return text.upper()
    elif emotion == "whispering":
        return text.lower()
    elif emotion == "questioning":
        # Ensure question mark at end
        if not text.endswith("?"):
            text = text.rstrip(".!") + "?"
        return text
    elif emotion == "excited":
        # Add emphasis
        if not text.endswith("!"):
            text = text.rstrip(".?") + "!"
        return text
    else:
        return text

def get_emotional_parameters(emotion: str) -> dict:
    """Get XTTS-v2 parameters for emotional variation (if supported)"""
    # Note: These parameters may not be directly supported by XTTS-v2
    # This is a placeholder for future emotional parameter support
    emotional_params = {
        "neutral": {},
        "happy": {"speed": 1.1},
        "sad": {"speed": 0.9},
        "angry": {"speed": 1.2},
        "excited": {"speed": 1.3},
        "questioning": {"speed": 1.0},
        "whispering": {"speed": 0.8},
        "shouting": {"speed": 1.4}
    }
    
    return emotional_params.get(emotion, {})

def apply_audio_emotional_effects(audio_path: str, emotion: str, job_id: str, index: int) -> str:
    """Apply post-processing effects to audio based on emotion"""
    try:
        import librosa
        import soundfile as sf
        
        # Load audio
        y, sr = librosa.load(audio_path)
        
        # Apply emotional effects
        if emotion == "whispering":
            # Reduce volume and add slight reverb effect
            y = y * 0.6
        elif emotion == "shouting":
            # Increase volume and add slight distortion
            y = y * 1.4
            y = np.clip(y, -1.0, 1.0)  # Prevent clipping
        elif emotion == "sad":
            # Slightly lower pitch
            y = librosa.effects.pitch_shift(y, sr=sr, n_steps=-1)
        elif emotion == "excited" or emotion == "happy":
            # Slightly higher pitch
            y = librosa.effects.pitch_shift(y, sr=sr, n_steps=1)
        
        # Save modified audio
        modified_path = f"/tmp/audio_modified_{job_id}_{index}.wav"
        sf.write(modified_path, y, sr)
        
        # Remove original and return modified path
        os.remove(audio_path)
        return modified_path
        
    except Exception as e:
        print(f"Failed to apply emotional effects: {e}")
        return audio_path  # Return original if processing fails

print("Coqui XTTS-v2 audio generation with emotional voice variation defined!")
```

```python
@app.route('/analyze', methods=['POST'])
def analyze_manga():
    """Start manga analysis"""
    try:
        data = request.json
        
        # Validate request
        if not data or 'pages' not in data:
            return jsonify({"error": "Missing pages data"}), 400
        
        pages_b64 = data['pages']
        character_bank = data.get('characterBank', {"images": [], "names": []})
        
        # Convert base64 images to numpy arrays
        chapter_pages = []
        for page_b64 in pages_b64:
            img = read_image_from_base64(page_b64)
            if img is not None:
                chapter_pages.append(img)
        
        if not chapter_pages:
            return jsonify({"error": "No valid images provided"}), 400
        
        # Create job
        job_id = create_job_id()
        analysis_jobs[job_id] = AnalysisJob(job_id)
        
        # Start analysis in background thread
        thread = threading.Thread(
            target=process_manga_analysis,
            args=(job_id, chapter_pages, character_bank)
        )
        thread.start()
        
        return jsonify({
            "job_id": job_id,
            "status": "started",
            "message": f"Analysis started for {len(chapter_pages)} pages"
        })
        
    except Exception as e:
        return jsonify({"error": str(e)}), 500
```

### 2. How Manga Data Flows to Colab:

**iOS App → Colab Flow:**
1. User opens a PDF manga chapter in the app
2. App automatically extracts PDF pages using PDFKit
3. Each page is converted to PNG image data
4. Images are base64-encoded for transmission

   - Array of base64-encoded page images
   - Character bank (if available for this manga)
6. Colab processes images with Magi model
7. Results sent back as JSON with dialogue and character data
8. App caches results locally for offline access

```
3. If not cached, app automatically:
   - Extracts PDF pages as images
   - Converts to base64
   - Sends to Colab endpoint
   - Shows subtle "Analyzing..." indicator
4. When complete, enables AI features (dialogue overlay, audio playback)

Colab Side (automatic):
1. Receives POST request with manga pages
2. Processes with Magi (text detection + OCR)
3. Returns JSON with dialogue and character data
4. Ready for audio generation if requested
```

## Single‑Cell Setup (All‑in‑One)

Paste the following into ONE Colab cell and run it. It installs deps, loads MagiV2 + XTTS‑v2, defines the API, starts ngrok, and runs the Flask server. When it prints the Public URL, copy it into the app’s AI Analysis Endpoint URL.

```python
# Single-cell, resilient to NumPy ABI issues. If it upgrades NumPy, the runtime will auto-restart.
import sys, subprocess, os, json, base64, uuid, time, threading
from io import BytesIO
from typing import List, Dict, Any

def ensure(pkg_args):
    subprocess.check_call([sys.executable, "-m", "pip", "install", "-q", "--upgrade", *pkg_args])

def ensure_numpy(version="1.26.4"):
    info = subprocess.run([sys.executable, "-m", "pip", "show", "numpy"], capture_output=True, text=True)
    if f"Version: {version}" not in info.stdout:
        ensure([f"numpy=={version}", "--force-reinstall"])
        print(f"Restarting runtime to load NumPy {version}…")
        os.kill(os.getpid(), 9)

# 1) Pin NumPy first to avoid ABI mismatch, then install other deps
ensure_numpy("1.26.4")
ensure(["torch", "torchvision", "torchaudio", "--index-url", "https://download.pytorch.org/whl/cu118"])
ensure(["transformers", "Pillow", "flask", "flask-cors", "requests", "tqdm", "pyngrok", "TTS", "librosa", "soundfile"]) 

import torch
import numpy as np
from PIL import Image
from transformers import AutoModel
from flask import Flask, request, jsonify
from flask_cors import CORS
from pyngrok import ngrok
from TTS.api import TTS

app = Flask(__name__)
CORS(app)

magi_model = None
tts_models: Dict[str, Any] = {}
analysis_jobs: Dict[str, Any] = {}
audio_jobs: Dict[str, Any] = {}

def initialize_magi():
    global magi_model
    print("Loading MagiV2 model…")
    magi_model = AutoModel.from_pretrained("ragavsachdeva/magiv2", trust_remote_code=True).cuda().eval()
    print("MagiV2 loaded")

def initialize_tts():
    global tts_models
    print("Loading XTTS‑v2…")
    try:
        tts_models['xtts_v2'] = TTS("tts_models/multilingual/multi-dataset/xtts_v2", gpu=True)
        print("XTTS‑v2 loaded")
    except Exception as e:
        print(f"XTTS‑v2 load failed: {e}")

def read_image_from_base64(b64: str):
    try:
        data = base64.b64decode(b64)
        from PIL import Image as _Image
        img = _Image.open(BytesIO(data)).convert("L").convert("RGB")
        return np.array(img)
    except Exception:
        return None

def convert_bbox(b):
    if not b or len(b) < 4:
        return {"x": 0, "y": 0, "width": 0, "height": 0}
    return {"x": float(b[0]), "y": float(b[1]), "width": float(b[2]-b[0]), "height": float(b[3]-b[1])}

def create_job_id():
    return str(uuid.uuid4())

class Job:
    def __init__(self, job_id: str):
        self.job_id = job_id
        self.status = "pending"
        self.progress = 0.0
        self.result = None
        self.error = None
        self.created_at = time.time()
    def to_dict(self):
        return {
            "job_id": self.job_id,
            "status": self.status,
            "progress": self.progress,
            "result": self.result,
            "error": self.error,
            "created_at": self.created_at,
        }

def process_manga_analysis(job_id: str, chapter_pages: List[np.ndarray], character_bank: Dict[str, Any]):
    job = analysis_jobs[job_id]
    try:
        job.status = "processing"; job.progress = 0.1
        character_images = []
        for img_b64 in character_bank.get("images", []):
            img = read_image_from_base64(img_b64)
            if img is not None:
                character_images.append(img)
        character_bank_processed = {"images": character_images, "names": character_bank.get("names", [])}
        job.progress = 0.2
        with torch.no_grad():
            per_page_results = magi_model.do_chapter_wide_prediction(
                chapter_pages, character_bank_processed, use_tqdm=True, do_ocr=True
            )
        job.progress = 0.8
        transcript = []
        pages_analysis = []
        for i, page_result in enumerate(per_page_results):
            speaker_name = {t: page_result["character_names"][c] for t, c in page_result["text_character_associations"]}
            text_regions = []
            for j, text in enumerate(page_result["ocr"]):
                if page_result["is_essential_text"][j]:
                    bbox = {}
                    if "text_bboxes" in page_result and j < len(page_result["text_bboxes"]):
                        bbox = convert_bbox(page_result["text_bboxes"][j])
                    text_regions.append({"id": j, "text": text, "boundingBox": bbox, "confidence": 1.0, "isEssential": True})
                    transcript.append({"pageIndex": i, "textId": j, "speaker": speaker_name.get(j, "unknown"), "text": text, "timestamp": None})
            character_detections = []
            for cidx, cname in enumerate(page_result["character_names"]):
                bbox = {}
                if "character_bboxes" in page_result and cidx < len(page_result["character_bboxes"]):
                    bbox = convert_bbox(page_result["character_bboxes"][cidx])
                character_detections.append({"id": cidx, "name": cname, "boundingBox": bbox, "confidence": 1.0})
            pages_analysis.append({
                "pageIndex": i,
                "textRegions": text_regions,
                "characterDetections": character_detections,
                "textCharacterAssociations": page_result["text_character_associations"],
            })
        job.result = {"pages": pages_analysis, "transcript": transcript, "analysisDate": time.time(), "version": "1.0"}
        job.status = "completed"; job.progress = 1.0
    except Exception as e:
        job.status = "failed"; job.error = str(e)

def detect_emotion_for_api(text: str) -> str:
    t = text.lower()
    if "!!!" in t or "!!!!" in t: return "shouting"
    if "!" in t:
        if any(k in t for k in ["no", "stop", "wait"]): return "angry"
        return "excited"
    if "?" in t: return "questioning"
    if "..." in t or ".." in t:
        if any(k in t for k in ["sigh", "sob"]): return "sad"
        return "whispering"
    if any(k in t for k in ["haha", "hehe", "laugh"]): return "happy"
    return "neutral"

def generate_audio_for_transcript(job_id: str, transcript: List[Dict], voice_settings: Dict[str, Any]):
    job = audio_jobs[job_id]
    try:
        job.status = "processing"; job.progress = 0.1
        tts_model = tts_models.get('xtts_v2')
        if not tts_model: raise Exception("XTTS‑v2 model not loaded")
        language = voice_settings.get("language", "en")
        character_voice_files = voice_settings.get("characterVoiceFiles", {})
        default_voice_file = voice_settings.get("defaultVoiceFile", None)
        audio_segments = []
        import librosa, soundfile as sf
        for i, d in enumerate(transcript):
            text = d.get("text", "").strip()
            if not text: continue
            emotion = d.get("emotion", detect_emotion_for_api(text))
            audio_path = f"/tmp/audio_{job_id}_{i}.wav"
            speaker = d.get("speaker", "")
            voice_b64 = character_voice_files.get(speaker, default_voice_file)
            if voice_b64:
                spk_path = f"/tmp/speaker_{job_id}_{i}.wav"
                with open(spk_path, "wb") as f: f.write(base64.b64decode(voice_b64))
                tts_model.tts_to_file(text=text, file_path=audio_path, speaker_wav=spk_path, language=language)
                os.remove(spk_path)
            else:
                kwargs = {"text": text, "file_path": audio_path, "language": language}
                if hasattr(tts_model, 'speakers') and tts_model.speakers:
                    kwargs["speaker"] = tts_model.speakers[0]
                tts_model.tts_to_file(**kwargs)
            y, sr = librosa.load(audio_path); duration = float(librosa.get_duration(y=y, sr=sr))
            with open(audio_path, "rb") as f: audio_b64 = base64.b64encode(f.read()).decode()
            os.remove(audio_path)
            audio_segments.append({"dialogueId": f"{d['pageIndex']}_{d['textId']}", "audioData": audio_b64, "duration": duration, "speaker": speaker, "text": d.get("text", "")})
            job.progress = 0.1 + 0.8 * (i + 1) / max(1, len(transcript))
        job.result = {"audioSegments": audio_segments}
        job.status = "completed"; job.progress = 1.0
    except Exception as e:
        job.status = "failed"; job.error = str(e)

@app.route('/health', methods=['GET'])
def health():
    return jsonify({
        "status": "healthy",
        "magi_loaded": magi_model is not None,
        "tts_models_loaded": len(tts_models),
        "xtts_v2_loaded": 'xtts_v2' in tts_models,
        "available_speakers": tts_models['xtts_v2'].speakers if 'xtts_v2' in tts_models and hasattr(tts_models['xtts_v2'], 'speakers') else []
    })

@app.route('/analyze', methods=['POST'])
def analyze():
    data = request.json or {}
    pages_b64 = data.get('pages', [])
    character_bank = data.get('characterBank', {"images": [], "names": []})
    chapter_pages = [img for img in (read_image_from_base64(b) for b in pages_b64) if img is not None]
    if not chapter_pages: return jsonify({"error": "No valid images provided"}), 400
    job_id = create_job_id(); analysis_jobs[job_id] = Job(job_id)
    t = threading.Thread(target=process_manga_analysis, args=(job_id, chapter_pages, character_bank))
    t.start()
    return jsonify({"job_id": job_id, "status": "started"})

@app.route('/status/<job_id>', methods=['GET'])
def status(job_id):
    if job_id not in analysis_jobs: return jsonify({"error": "Job not found"}), 404
    return jsonify(analysis_jobs[job_id].to_dict())

@app.route('/result/<job_id>', methods=['GET'])
def result(job_id):
    if job_id not in analysis_jobs: return jsonify({"error": "Job not found"}), 404
    job = analysis_jobs[job_id]
    if job.status != "completed":
        return jsonify({"error": "Job not completed", "status": job.status, "progress": job.progress}), 400
    return jsonify(job.result)

@app.route('/audio', methods=['POST'])
def audio():
    data = request.json or {}
    transcript = data.get('transcript', [])
    voice_settings = data.get('voiceSettings', {"language": "en", "characterVoiceFiles": {}, "defaultVoiceFile": None})
    job_id = create_job_id(); audio_jobs[job_id] = Job(job_id)
    t = threading.Thread(target=generate_audio_for_transcript, args=(job_id, transcript, voice_settings))
    t.start()
    return jsonify({"job_id": job_id, "status": "started"})

@app.route('/audio/status/<job_id>', methods=['GET'])
def audio_status(job_id):
    if job_id not in audio_jobs: return jsonify({"error": "Job not found"}), 404
    return jsonify(audio_jobs[job_id].to_dict())

@app.route('/audio/result/<job_id>', methods=['GET'])
def audio_result(job_id):
    if job_id not in audio_jobs: return jsonify({"error": "Job not found"}), 404
    job = audio_jobs[job_id]
    if job.status != "completed":
        return jsonify({"error": "Job not completed", "status": job.status, "progress": job.progress}), 400
    return jsonify(job.result)
```
